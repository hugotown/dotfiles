# draft-ptb Extension Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a pi extension that orchestrates a brainstorm → spec → plan → execute workflow using a state machine, terminate tools for structured LLM output, TUI for user interaction, and context stripping for ~92% token savings.

**Architecture:** State machine (pure reducer) drives phases. Each creative phase uses a terminate tool so the LLM produces structured output and stops. The `agent_end` event handler dispatches TUI interactions and triggers the next phase via `drivePhase()`. A phase marker + context filter ensures each LLM turn sees only its own prompt (no accumulated history). Model switching routes creative work (design) to Opus and structured work (research, plan) to Sonnet.

**Tech Stack:** TypeScript, pi Extension API (`@earendil-works/pi-coding-agent`), TypeBox schemas, pi TUI (`ctx.ui.*`)

```typescript
// ~/.pi/agent/extensions/draft-ptb/state.ts

export type Phase =
  | "IDLE"
  | "GATHERING_CONTEXT"
  | "RESEARCH"
  | "APPROACHES"
  | "DESIGN"
  | "PLAN"
  | "COMPLETE";

export interface Question {
  question: string;
  options?: string[];
}

export interface Answer {
  question: string;
  answer: string;
}

export interface Approach {
  name: string;
  description: string;
  tradeoffs: string;
}

export interface DraftState {
  phase: Phase;
  idea: string;
  compressedContext: string;
  questions: Question[];
  answers: Answer[];
  approaches: Approach[];
  recommendation: string;
  chosenApproach: Approach | null;
  specTitle: string;
  spec: string;
  specPath: string | null;
  plan: string;
  planPath: string | null;
  originalModel: { provider: string; id: string } | null;
  allToolNames: string[];
}

export type DraftEvent =
  | { type: "START" }
  | { type: "CONTEXT_READY"; compressedContext: string }
  | { type: "QUESTIONS_RECEIVED"; questions: Question[] }
  | { type: "ANSWERS_COLLECTED"; answers: Answer[] }
  | { type: "APPROACHES_RECEIVED"; approaches: Approach[]; recommendation: string }
  | { type: "APPROACH_CHOSEN"; approach: Approach }
  | { type: "SPEC_RECEIVED"; title: string; spec: string }
  | { type: "SPEC_APPROVED"; specPath: string }
  | { type: "SPEC_REVISION_REQUESTED"; feedback: string }
  | { type: "PLAN_RECEIVED"; plan: string }
  | { type: "PLAN_SAVED"; planPath: string }
  | { type: "RESET" };

export function createInitialState(idea: string): DraftState {
  return {
    phase: "IDLE",
    idea,
    compressedContext: "",
    questions: [],
    answers: [],
    approaches: [],
    recommendation: "",
    chosenApproach: null,
    specTitle: "",
    spec: "",
    specPath: null,
    plan: "",
    planPath: null,
    originalModel: null,
    allToolNames: [],
  };
}

/**
 * Pure reducer. Every transition is guarded by the current phase.
 * RESET is honored from any phase.
 */
export function transition(state: DraftState, event: DraftEvent): DraftState {
  if (event.type === "RESET") {
    return { ...state, phase: "IDLE" };
  }

  switch (state.phase) {
    case "IDLE":
      if (event.type === "START") return { ...state, phase: "GATHERING_CONTEXT" };
      return state;

    case "GATHERING_CONTEXT":
      if (event.type === "CONTEXT_READY")
        return { ...state, phase: "RESEARCH", compressedContext: event.compressedContext };
      return state;

    case "RESEARCH":
      if (event.type === "QUESTIONS_RECEIVED")
        return { ...state, questions: event.questions };
      if (event.type === "ANSWERS_COLLECTED")
        return { ...state, phase: "APPROACHES", answers: event.answers };
      return state;

    case "APPROACHES":
      if (event.type === "APPROACHES_RECEIVED")
        return { ...state, approaches: event.approaches, recommendation: event.recommendation };
      if (event.type === "APPROACH_CHOSEN")
        return { ...state, phase: "DESIGN", chosenApproach: event.approach };
      return state;

    case "DESIGN":
      if (event.type === "SPEC_RECEIVED")
        return { ...state, specTitle: event.title, spec: event.spec };
      if (event.type === "SPEC_APPROVED")
        return { ...state, phase: "PLAN", specPath: event.specPath };
      if (event.type === "SPEC_REVISION_REQUESTED")
        return state; // Stay in DESIGN, re-trigger with feedback
      return state;

    case "PLAN":
      if (event.type === "PLAN_RECEIVED")
        return { ...state, plan: event.plan };
      if (event.type === "PLAN_SAVED")
        return { ...state, phase: "COMPLETE", planPath: event.planPath };
      return state;

    default:
      return state;
  }
}
```

```typescript
// ~/.pi/agent/extensions/draft-ptb/context-builder.ts

import type { ExtensionAPI } from "@earendil-works/pi-coding-agent";

/**
 * Reads project context mechanically. No LLM tokens spent.
 * Returns a compressed string (~1.5K tokens) with project info.
 */
export async function buildProjectContext(pi: ExtensionAPI, cwd: string): Promise<string> {
  const sections: string[] = [];

  // 1. Directory tree (depth 2, excluding noise)
  const tree = await pi.exec("eza", ["--tree", "--level=2", "-I", "node_modules|.git|dist|build|coverage"], { cwd });
  if (tree.code === 0 && tree.stdout.trim()) {
    sections.push(`## Project Structure\n\`\`\`\n${truncate(tree.stdout, 2000)}\n\`\`\``);
  }

  // 2. Package.json — extract only what matters
  const pkg = await pi.exec("cat", ["package.json"], { cwd });
  if (pkg.code === 0) {
    try {
      const parsed = JSON.parse(pkg.stdout);
      const summary = {
        name: parsed.name,
        description: parsed.description,
        scripts: Object.keys(parsed.scripts ?? {}),
        deps: Object.keys(parsed.dependencies ?? {}),
        devDeps: Object.keys(parsed.devDependencies ?? {}),
      };
      sections.push(`## Tech Stack\n\`\`\`json\n${JSON.stringify(summary, null, 2)}\n\`\`\``);
    } catch {
      // Not a node project, skip
    }
  }

  // 3. Cargo.toml for Rust projects
  const cargo = await pi.exec("cat", ["Cargo.toml"], { cwd });
  if (cargo.code === 0 && cargo.stdout.trim()) {
    sections.push(`## Cargo.toml (excerpt)\n\`\`\`toml\n${truncate(cargo.stdout, 1000)}\n\`\`\``);
  }

  // 4. Recent commits (context of recent work)
  const git = await pi.exec("git", ["log", "--oneline", "-10"], { cwd });
  if (git.code === 0 && git.stdout.trim()) {
    sections.push(`## Recent Commits\n\`\`\`\n${git.stdout.trim()}\n\`\`\``);
  }

  // 5. README first 50 lines
  const readme = await pi.exec("head", ["-50", "README.md"], { cwd });
  if (readme.code === 0 && readme.stdout.trim()) {
    sections.push(`## README (excerpt)\n${truncate(readme.stdout, 1000)}`);
  }

  if (sections.length === 0) {
    return "(No project context could be gathered from the current directory)";
  }

  return sections.join("\n\n");
}

function truncate(text: string, maxChars: number): string {
  if (text.length <= maxChars) return text;
  return text.slice(0, maxChars) + "\n... (truncated)";
}
```

```typescript
// ~/.pi/agent/extensions/draft-ptb/orchestrator.ts

import type { ExtensionAPI, ExtensionContext } from "@earendil-works/pi-coding-agent";
import { PHASE_CONFIG } from "./config.ts";
import type { DraftState, Phase } from "./state.ts";

export const STATE_ENTRY = "draft-ptb-state";
const PHASE_MARKER = "draft-ptb-phase-marker";

/** Persist the full state after every transition. */
export function persist(pi: ExtensionAPI, state: DraftState): void {
  pi.appendEntry(STATE_ENTRY, state);
}

/** Restore the latest persisted state from the session branch, or null. */
export function restore(ctx: ExtensionContext): DraftState | null {
  const entries = ctx.sessionManager.getBranch();
  for (let i = entries.length - 1; i >= 0; i--) {
    const e = entries[i] as { type?: string; customType?: string; data?: DraftState };
    if (e.type === "custom" && e.customType === STATE_ENTRY && e.data) {
      return e.data;
    }
  }
  return null;
}

/** Apply model + tool scope for a phase. Returns false if model unavailable. */
export async function applyPhaseConfig(
  pi: ExtensionAPI,
  ctx: ExtensionContext,
  phase: Phase,
): Promise<boolean> {
  const config = PHASE_CONFIG[phase];
  if (!config) return true; // non-LLM phase (IDLE, GATHERING_CONTEXT, COMPLETE)
  const model = ctx.modelRegistry.find(config.model.provider, config.model.id);
  if (!model) {
    ctx.ui.notify(`Model ${config.model.provider}/${config.model.id} not found.`, "error");
    return false;
  }
  const ok = await pi.setModel(model);
  if (!ok) {
    ctx.ui.notify(`No API key for ${config.model.provider}/${config.model.id}.`, "error");
    return false;
  }
  pi.setActiveTools(config.tools);
  return true;
}

/** Capture session defaults (model + tools) so they can be restored later. */
export function captureDefaults(
  pi: ExtensionAPI,
  ctx: ExtensionContext,
  state: DraftState,
): DraftState {
  const all = pi.getAllTools().map((t) => t.name);
  const model = ctx.model ? { provider: ctx.model.provider, id: ctx.model.id } : null;
  return { ...state, originalModel: model, allToolNames: all };
}

/** Restore the original model + tools (on COMPLETE or RESET). */
export async function restoreDefaults(
  pi: ExtensionAPI,
  ctx: ExtensionContext,
  state: DraftState,
): Promise<void> {
  if (state.allToolNames.length) pi.setActiveTools(state.allToolNames);
  if (state.originalModel) {
    const m = ctx.modelRegistry.find(state.originalModel.provider, state.originalModel.id);
    if (m) await pi.setModel(m);
  }
}

/**
 * Inject a hidden phase marker, then send the phase prompt as a user message.
 * The marker lets the context filter trim everything from prior phases.
 */
export function drivePhase(pi: ExtensionAPI, prompt: string): void {
  pi.sendMessage({ customType: PHASE_MARKER, content: "", display: false }, { triggerTurn: false });
  pi.sendUserMessage(prompt);
}

/**
 * Context filter: keep only messages at/after the most recent phase marker.
 * This prevents history accumulation — each LLM turn sees only its own prompt.
 */
export function filterContext(messages: unknown[]): unknown[] {
  let lastMarker = -1;
  for (let i = messages.length - 1; i >= 0; i--) {
    const m = messages[i] as { customType?: string };
    if (m.customType === PHASE_MARKER) {
      lastMarker = i;
      break;
    }
  }
  if (lastMarker === -1) return messages;
  return messages.slice(lastMarker + 1);
}
```

```typescript
// ~/.pi/agent/extensions/draft-ptb/index.ts

import type { ExtensionAPI, ExtensionContext } from "@earendil-works/pi-coding-agent";
import { Type } from "typebox";
import {
  createInitialState,
  transition,
  type DraftEvent,
  type DraftState,
  type Question,
  type Approach,
} from "./state.ts";
import {
  applyPhaseConfig,
  captureDefaults,
  drivePhase,
  filterContext,
  persist,
  restore,
  restoreDefaults,
} from "./orchestrator.ts";
import { buildProjectContext } from "./context-builder.ts";

export default function draftPtb(pi: ExtensionAPI): void {
  let state: DraftState | null = null;

  // ─── Command: /draft-ptb <idea> ───────────────────────────────────────

  pi.registerCommand("draft-ptb", {
    description: "Orchestrated brainstorm → design → plan workflow (token-efficient)",
    handler: async (args, ctx) => {
      if (!ctx.hasUI) {
        ctx.ui.notify("draft-ptb requires an interactive UI.", "error");
        return;
      }
      const idea = args.trim();
      if (!idea) {
        ctx.ui.notify("Usage: /draft-ptb <describe your idea>", "warning");
        return;
      }

      state = captureDefaults(pi, ctx, createInitialState(idea));
      await advance(pi, ctx, { type: "START" });
    },
  });

  // ─── Terminate Tools ──────────────────────────────────────────────────

  pi.registerTool({
    name: "draft_ptb_research",
    label: "Draft: Research",
    description:
      "Submit clarifying questions about the user's idea. You MUST call this tool with 3-5 questions.",
    parameters: Type.Object({
      questions: Type.Array(
        Type.Object({
          question: Type.String({ description: "The question text" }),
          options: Type.Optional(
            Type.Array(Type.String(), { description: "Multiple choice options if applicable" }),
          ),
        }),
        { description: "3-5 clarifying questions to understand the idea better" },
      ),
    }),
    async execute(_toolCallId, params) {
      if (!state) throw new Error("No active draft-ptb workflow");
      state = transition(state, {
        type: "QUESTIONS_RECEIVED",
        questions: params.questions as Question[],
      });
      return {
        content: [{ type: "text", text: `Generated ${params.questions.length} questions.` }],
        details: { questions: params.questions },
        terminate: true,
      };
    },
  });

  pi.registerTool({
    name: "draft_ptb_approaches",
    label: "Draft: Approaches",
    description:
      "Submit 2-3 implementation approaches with trade-offs. You MUST call this tool.",
    parameters: Type.Object({
      approaches: Type.Array(
        Type.Object({
          name: Type.String({ description: "Short approach name" }),
          description: Type.String({ description: "What this approach does" }),
          tradeoffs: Type.String({ description: "Pros/cons" }),
        }),
        { description: "2-3 distinct implementation approaches" },
      ),
      recommendation: Type.String({ description: "Which approach you recommend and why" }),
    }),
    async execute(_toolCallId, params) {
      if (!state) throw new Error("No active draft-ptb workflow");
      state = transition(state, {
        type: "APPROACHES_RECEIVED",
        approaches: params.approaches as Approach[],
        recommendation: params.recommendation,
      });
      return {
        content: [{ type: "text", text: `Proposed ${params.approaches.length} approaches.` }],
        details: { approaches: params.approaches, recommendation: params.recommendation },
        terminate: true,
      };
    },
  });

  pi.registerTool({
    name: "draft_ptb_design",
    label: "Draft: Design",
    description:
      "Submit the complete design spec. You MUST call this tool with the full spec in markdown.",
    parameters: Type.Object({
      title: Type.String({ description: "Spec title" }),
      spec: Type.String({ description: "Complete design spec in markdown" }),
    }),
    async execute(_toolCallId, params) {
      if (!state) throw new Error("No active draft-ptb workflow");
      state = transition(state, {
        type: "SPEC_RECEIVED",
        title: params.title,
        spec: params.spec,
      });
      return {
        content: [{ type: "text", text: `Design spec: "${params.title}"` }],
        details: { title: params.title, specLength: params.spec.length },
        terminate: true,
      };
    },
  });

  pi.registerTool({
    name: "draft_ptb_plan",
    label: "Draft: Plan",
    description:
      "Submit the complete implementation plan. You MUST call this tool with the full plan in markdown.",
    parameters: Type.Object({
      plan: Type.String({ description: "Complete implementation plan in markdown" }),
    }),
    async execute(_toolCallId, params) {
      if (!state) throw new Error("No active draft-ptb workflow");
      state = transition(state, { type: "PLAN_RECEIVED", plan: params.plan });
      return {
        content: [{ type: "text", text: "Implementation plan received." }],
        details: { planLength: params.plan.length },
        terminate: true,
      };
    },
  });

  // ─── Context Filter ───────────────────────────────────────────────────

  pi.on("context", async (event) => {
    if (!state || state.phase === "IDLE" || state.phase === "COMPLETE") return;
    return { messages: filterContext(event.messages as unknown[]) as typeof event.messages };
  });

  // ─── Agent End Dispatcher ─────────────────────────────────────────────

  pi.on("agent_end", async (_event, ctx) => {
    if (!state || state.phase === "IDLE" || state.phase === "COMPLETE") return;

    switch (state.phase) {
      case "RESEARCH": {
        // Questions have been stored by the tool. Now ask them via TUI.
        if (state.questions.length === 0) return;

        const answers: { question: string; answer: string }[] = [];
        for (const q of state.questions) {
          let answer: string | undefined;
          if (q.options?.length) {
            answer = await ctx.ui.select(q.question, q.options);
          } else {
            answer = (await ctx.ui.input(q.question)) ?? "";
          }
          answers.push({ question: q.question, answer: answer ?? "(skipped)" });
        }

        await advance(pi, ctx, { type: "ANSWERS_COLLECTED", answers });
        break;
      }

      case "APPROACHES": {
        // Approaches have been stored by the tool. Now let user pick.
        if (state.approaches.length === 0) return;

        ctx.ui.notify(`Recommendation: ${state.recommendation}`, "info");

        const choice = await ctx.ui.select(
          "Pick an approach:",
          state.approaches.map((a) => `${a.name}: ${a.description}`),
        );

        const chosenIndex = state.approaches.findIndex(
          (a) => choice?.startsWith(a.name),
        );
        const chosen = state.approaches[chosenIndex >= 0 ? chosenIndex : 0];

        await advance(pi, ctx, { type: "APPROACH_CHOSEN", approach: chosen });
        break;
      }

      case "DESIGN": {
        // Spec has been stored by the tool. Let user review.
        if (!state.spec) return;

        const feedback = await ctx.ui.editor(
          "Review the spec (edit to provide feedback, or leave unchanged to approve):",
          state.spec,
        );

        const approved = !feedback?.trim() || feedback.trim() === state.spec.trim();

        if (approved) {
          // Save spec to disk
          const specPath = await saveSpec(pi, ctx, state);
          await advance(pi, ctx, { type: "SPEC_APPROVED", specPath });
        } else {
          // Re-trigger DESIGN with feedback
          await advance(pi, ctx, { type: "SPEC_REVISION_REQUESTED", feedback: feedback! });
        }
        break;
      }

      case "PLAN": {
        // Plan has been stored by the tool. Save it.
        if (!state.plan) return;

        const planPath = await savePlan(pi, ctx, state);
        await advance(pi, ctx, { type: "PLAN_SAVED", planPath });
        break;
      }
    }
  });

  // ─── Session Start — Restore State ────────────────────────────────────

  pi.on("session_start", async (event, ctx) => {
    if (event.reason !== "resume" && event.reason !== "reload") return;
    const restored = restore(ctx);
    if (restored && restored.phase !== "IDLE" && restored.phase !== "COMPLETE") {
      state = restored;
      ctx.ui.notify(`draft-ptb resumed at phase: ${state.phase}`, "info");
    }
  });

  // ─── Internal Helpers ─────────────────────────────────────────────────

  async function advance(api: ExtensionAPI, ctx: ExtensionContext, event: DraftEvent): Promise<void> {
    if (!state) return;
    state = transition(state, event);
    persist(api, state);
    await driveCurrentPhase(api, ctx);
  }

  async function driveCurrentPhase(api: ExtensionAPI, ctx: ExtensionContext): Promise<void> {
    if (!state) return;

    switch (state.phase) {
      case "GATHERING_CONTEXT": {
        ctx.ui.setStatus("draft-ptb", "📦 gathering context...");
        const compressed = await buildProjectContext(api, ctx.cwd);
        await advance(api, ctx, { type: "CONTEXT_READY", compressedContext: compressed });
        break;
      }

      case "RESEARCH": {
        ctx.ui.setStatus("draft-ptb", "🔬 research");
        const ok = await applyPhaseConfig(api, ctx, "RESEARCH");
        if (!ok) { await advance(api, ctx, { type: "RESET" }); return; }

        drivePhase(api, buildResearchPrompt(state));
        break;
      }

      case "APPROACHES": {
        ctx.ui.setStatus("draft-ptb", "🎯 approaches");
        const ok = await applyPhaseConfig(api, ctx, "APPROACHES");
        if (!ok) { await advance(api, ctx, { type: "RESET" }); return; }

        drivePhase(api, buildApproachesPrompt(state));
        break;
      }

      case "DESIGN": {
        ctx.ui.setStatus("draft-ptb", "✏️ design");
        const ok = await applyPhaseConfig(api, ctx, "DESIGN");
        if (!ok) { await advance(api, ctx, { type: "RESET" }); return; }

        // If there's existing spec + revision feedback, include it
        const hasRevision = state.spec && state.answers.length > 0;
        drivePhase(api, buildDesignPrompt(state));
        break;
      }

      case "PLAN": {
        ctx.ui.setStatus("draft-ptb", "📋 plan");
        const ok = await applyPhaseConfig(api, ctx, "PLAN");
        if (!ok) { await advance(api, ctx, { type: "RESET" }); return; }

        drivePhase(api, buildPlanPrompt(state));
        break;
      }

      case "COMPLETE": {
        ctx.ui.setStatus("draft-ptb", undefined);
        await restoreDefaults(api, ctx, state);

        ctx.ui.notify(
          `✅ draft-ptb complete!\n  Spec: ${state.specPath}\n  Plan: ${state.planPath}`,
          "info",
        );

        // Offer execution
        const choice = await ctx.ui.select("What next?", [
          "Execute the plan (subagent-driven)",
          "Done for now",
        ]);

        if (choice?.startsWith("Execute")) {
          pi.sendUserMessage(
            `Execute the plan at ${state.planPath} using subagent-driven-development.\n` +
            `- Dispatch fresh subagent per task\n` +
            `- Two-stage review (spec compliance + code quality)\n` +
            `- Don't stop between tasks`,
          );
        }
        break;
      }

      case "IDLE": {
        ctx.ui.setStatus("draft-ptb", undefined);
        await restoreDefaults(api, ctx, state);
        break;
      }
    }
  }
}
```

## Notes

1. **No npm dependencies needed.** `typebox` and `@earendil-works/pi-coding-agent` types are resolved by pi's jiti loader from the global installation.

2. **Token budget per phase:** Each LLM turn receives ~5K tokens of input (compressed project context + idea + Q&A). With 4 LLM turns total, the workflow uses ~20K tokens vs ~262K tokens for a traditional 12-15 turn conversation. That's ~92% savings.

3. **Resumability:** State is persisted after every transition via `pi.appendEntry()`. On `session_start` (resume/reload), the state is restored. However, the workflow cannot auto-resume mid-phase (e.g., if interrupted during the DESIGN LLM call). The user would need to re-run the command.

4. **Model availability:** If the configured model (e.g., Opus) is not available (no API key), the workflow resets cleanly with a notification. Fallback to a different model could be added later.

5. **The `context` filter is critical.** Without it, each successive LLM turn would accumulate the full history of prior turns, growing the context window linearly. The phase marker approach ensures clean ~5K context per turn.
